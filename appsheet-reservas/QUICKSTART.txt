📦 PASO 1: DESPLEGAR API (3 minutos)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  $ cd appsheet-reservas
  $ npm install
  $ vercel --prod

  ✅ Guarda tu URL: https://tu-proyecto.vercel.app

📱 PASO 3: CREAR APP EN APPSHEET (3 minutos)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  1. Ve a appsheet.com
  2. Create → App → Start with your own data
  3. Nombre: "Reservas Restaurante"
  4. Choose data → API → REST API

  Configuración:
  ┌─────────────────┬──────────────────────────────────────┐
  │ Connection Name │ ReservasAPI                          │
  │ Base URL        │ https://tu-proyecto.vercel.app       │
  │ Authentication  │ Custom Headers                       │
  │ Header Name     │ X-Api-Key                            │
  │ Header Value    │ appsheet-cronos-2024                 │
  └─────────────────┴──────────────────────────────────────┘


📊 PASO 4: CONFIGURAR TABLA (2 minutos)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  Data → Tables → + New Table

  ┌─────────────────┬──────────────────┐
  │ Table Name      │ Reservas         │
  │ Endpoint        │ /api/reservations│
  │ Method          │ GET              │
  │ Key Column      │ id_reserva       │
  └─────────────────┴──────────────────┘

  Configurar CRUD:
  
  CREATE (POST):
  Endpoint: /api/reservations
  Body:
  {
    "nom_persona_reserva": "<<[nom_persona_reserva]>>",
    "telefon": "<<[telefon]>>",
    "data_reserva": "<<[data_reserva]>>",
    "num_persones": <<[num_persones]>>,
    "observacions": "<<[observacions]>>",
    "status": "pending"
  }

  UPDATE (PUT):
  Endpoint: /api/reservations?id=<<[id_reserva]>>
  
  DELETE:
  Endpoint: /api/reservations?id=<<[id_reserva]>>


📅 PASO 5: CREAR CALENDARIO (2 minutos)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  UX → Views → + New View

  ┌─────────────────┬──────────────────────────────────────────────────┐
  │ View name       │ Calendario                                       │
  │ View type       │ calendar                                         │
  │ Primary view    │ Yes                                              │
  │ For this data   │ Reservas                                         │
  └─────────────────┴──────────────────────────────────────────────────┘

  Start: [data_reserva]
  End:   [data_reserva] + "02:00:00"
  Label: [nom_persona_reserva] & " (" & TEXT([num_persones]) & " personas)"
  Color: SWITCH([status],"confirmed","#4CAF50","pending","#FFA500",
                 "cancelled","#F44336","completed","#2196F3","#808080")

💡 TIPS
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  • Usa formato de fecha: YYYY-MM-DD HH:MM:SS
  • Número de personas: 1-20
  • Campos obligatorios: nom_persona_reserva, telefon, data_reserva, num_persones
  • Status válidos: pending, confirmed, cancelled, completed


🐛 PROBLEMAS COMUNES
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  "API Key inválida"
  → Verifica header X-Api-Key en AppSheet

  "No se cargan datos"
  → Verifica variables de entorno en Vercel
  → Redesplegar después de añadir variables

  "Formato de fecha inválido"
  → Usa formato: 2024-10-25 20:00:00 (sin 'T' ni 'Z')
